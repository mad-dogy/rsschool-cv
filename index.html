void BubbleSort(Student* list, int size) {
    for (int i = 0; i < size; i++) {             
        for (int j = 0; j < size - 1; j++) {
            if (list[j].score < list[j + 1].score) {
                Student temp = list[j];
                list[j] = list[j + 1];
                list[j + 1] = temp;
            }
        }
    }
}
